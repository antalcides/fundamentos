\section{Axiomas de la Teoría de Conjuntos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Para intentar formalizar la Teoría de Conjuntos suupondremos que
tenemos algunos \emph{objetos primitivos} como números o 
letras y llamamos \emph{conjunto} a una colección de objetos
primitivos u otros conjuntos si ésta es construida a partir de los 
siguientes axiomas. 

\begin{ax}{\bf Esquema de Comprensión.}
Si $A$ es un conjunto y $p$ es una propiedad, la colección de los
elementos en $A$ que cumplen la propiedad $p$ es un conjunto al que
denotamos $\{a\in A\mid p(a)\}$.
\end{ax}

\begin{ax}{\bf Axioma del Par.}
Si $A$ y $B$ son dos objetos, es decir, objetos primitivos o
conjuntos, la colección que tiene por elementos a $A$ y $B$ es un
conjunto al que denotamos $\{A,B\}$.
\end{ax}

\begin{ax}{\bf Axioma de la Unión.}
Si $A$ es un conjunto de conjuntos, es decir, un conjunto cuyos
elementos son conjuntos, la colección cuyos elementos son los
elementos de los elementos de $A$ es un conjunto al que denotamos 
$\bigcup A$. 
\end{ax}

\begin{ax}{\bf Axioma del Conjunto Potencia.}
Si $A$ es un conjunto, la colección que tiene por elementos a los
subconjuntos de $A$ es un conjunto al que denotamos $\mathcal{P}(A)$.
\end{ax}

\begin{ax}{\bf Axioma del Infinito}
La colección que tiene por elementos a los números naturales
$0,1,2,\dots$ es un conjunto al que denotamos $\mathbb{N}$.
\end{ax}

\begin{ax}{\bf Axioma de Reemplazo.}
Si $I$ es un conjunto y para cada elemento $i$ de $I$ se tiene un
conjunto $A_i$, la colección que tiene por elementos a los conjuntos
$A_i$ para toda $i$ en $I$ es un conjunto al que denotamos
$\{A_i\}_{i\in I}$ o $\{A_i\mid i\in I\}$.

A los conjuntos construidos de esta manera los llamamos
\emph{familias de conjuntos} y al conjunto $I$ lo llamamos el
\emph{conjunto de índices de la familia}.
\end{ax}

Observa que dos conjuntos son iguales si y sólo si como colecciones
son iguales, es decir, si y sólo si tienen los mismos elementos.

\begin{de}
Si $A$ y $B$ son dos conjuntos, decimos que $A$ está contenido en $B$
y escribimos $A\subset B$ si todo elemento de $A$ es también un
elemento de $B$.

En particular, dos conjuntos $A$ y $B$ son iguales si y sólo si $A$
está contenido en $B$ y $B$ está contenido en $A$.
\end{de}

\section{Existencia de algunos conjuntos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{te}
La colección de todos los conjuntos no es conjunto.
\end{te}

\begin{te}
La colección que no tiene elementos es un conjunto al que denotamos 
$\emptyset$ y llamamos \emph{conjunto vacío}. 
\end{te}

Este conjunto tiene la siguiente propiedad.

\begin{te}
El conjunto vacío está contenido en cualquier otro conjunto.
\end{te}

\begin{te}
Si $x$ es un objeto, la colección que tiene como único elemento a $x$
es un conjunto al que denotamos $\{x\}$ y llamamos \emph{singulete}. 
\end{te}

Más generalemente.

\begin{te}
Si $a_1$, $a_2$, $\dots$, $a_n$ son $n$ objetos, la colección
que tiene por elemenetos a los objetos $a_1$, $a_2$, $\dots$,
$a_n$ es un conjunto al que denotamos $\{a_1,a_2,\dots,a_n\}$. 
\end{te}

Con esto podemos decir que las colecciones con un número finito de
objetos son conjuntos.

\section{Algunas propiedades del Conjunto Potencia}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\def\po#1{\mathcal{P}(#1)}

Recuerda que si $A$ es un conjunto, la colección que tiene por
elementos a los subconjuntos de $A$ es un conjunto al que denotamos
$\po{A}$ y llamamos el conjunto potencia de $A$. Por ejemplo,
\begin{enumerate}
\item $\po{\emptyset}=\{\emptyset\}$ 
\item $\po{\{a\}}=\{\emptyset, \{a\}\}$
\item $\po{\{a,b\}}=\{\emptyset, \{a\},\{b\},\{a,b\}\}$
\end{enumerate}

Nota que el conjunto potencia de un conjunto $A$ siempre tiene como
elementos al conjunto vacío $\emptyset$ y a $A$ mismo. En particular, 
$\po{A}$ siempre es diferente del vacío.

\begin{te}
Sean $A$ y $B$ dos conjuntos. Si $A\subset B$ entonces
$\po{A}\subset\po{B}$ 
\end{te}

\begin{te}
Si $A$ es un conjunto, el conjunto potencia de $A$ es un conjunto de
conjuntos y el conjunto unión del conjunto potencia de $A$ es igual a
$A$, es decir, $\bigcup \po{A}=A$.
\end{te}

\section{Uniones e Intersecciones finitas de conjuntos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{te}
Si $A$ y $B$ son dos conjuntos, la colección que tiene por elementos a
los elementos que pertenecen al mismo tiempo a $A$ y a $B$ es un
conjunto al que denotamos  $A\cap B$ y llamamos \emph{la intersección
de $A$ y de $B$.} 
\end{te}

\begin{te}
Si $A$ y $B$ son dos conjuntos, la colección que tiene por elementos a
los elementos que pertenecen al conjunto $A$ o al conjunto $B$ (o a
ambos) es un conjunto al que denotamos  $A\cup B$ y llamamos \emph{la 
unión de $A$ y de $B$.} 
\end{te}

Por ejemplo, si $A=\{0,1,2,3,4,5\}$ y $B=\{0,2,4,6\}$ entonces
la intersección de $A$ y $B$ es igual a $A\cap B=\{0,2,4\}$ y la unión
de $A$ y $B$ es igual a $A\cup B=\{0,1,2,3,4,5,6\}$ 

Algunas propiedades de la unión y la intersección son las siguientes.

\begin{te}
Si $A$, $B$ y $C$ son conjuntos, entonces:
\begin{enumerate}
\item $A\cap A=A$ y $A\cup A=A$ \hfill {\sc (Idempotencia)} 
\item $A\cap B=B\cap A$ y $A\cup B=B\cup A$ \hfill {\sc (Conmutatividad)}
\item $(A\cap B)\cap C=A\cap(B\cap C)$ y $(A\cup B)\cup C=A\cup(B\cup
C)$ \hfill {\sc (Asociatividad)}
\item $C\cup (A\cap B)=(C\cup A)\cap (C\cup B)$ y
$C\cap (A\cup B)=(C\cap A)\cup (C\cap B)$ \hfill {\sc (Distributividad)}
\item  $A\cap \emptyset= \emptyset$ y $A\cup \emptyset =A$
\end{enumerate}
\end{te}

\section{Diferencias de conjuntos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{te}
Si $A$ y $B$ son dos conjuntos, la colección que tiene por elementos a
los elementos que pertenecen a $A$ y no pertenecen a $B$ es un
conjunto al que denotamos $A\backslash B$ y llamamos \emph{la
diferencia de $A$ y $B$}. 
\end{te}

Por ejemplo, si $A=\{0,1,2,3,4,5\}$ y $B=\{0,2,4,6\}$ entonces
la difernecia de $A$ y $B$ es igual a $A\backslash B=\{1,3,5\}$ y la
diferencia de $B$ y $A$ es igual a $B\backslash A =\{6\}$.

\newcommand{\dif}{\backslash}

\begin{te}
Si $A$ es un conjunto, entonces 
\begin{enumerate}
\item $A\dif \emptyset = A$ 
\item $\emptyset\dif A= \emptyset$   
\item $A\dif A=\emptyset$
\end{enumerate}
\end{te}

\section{Producto Cartesiano de Conjuntos}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{de}
Si $a$ y $b$ son dos objetos, definimos la \emph{pareja ordenada de $a$
y $b$} como el conjunto $\{\{a\},\{a,b\}\}$ y lo denotamos $(a,b)$.
\end{de}

\begin{te}
Sean $a$, $b$, $\alpha$ y $\beta$ objetos. Entonces,
$(a,b)=(\alpha,\beta)$ si y sólo si $a=\alpha$ y $b=\beta$.
\end{te}

\begin{te}
Si $A$ y $B$ son dos conjuntos, la colección de todas las parejas
ordenadas $(a,b)$ donde $a$ es un elemento de $A$ y $b$ es un elemento
de $B$ es un conjunto al que denotamos $A\times B$ y llamamos \emph{el
producto cartesiano de $A$ y $B$}.
\end{te}

\section{Uniones e Intersecciones de Familias}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{te}
Si $\{A_i\}_{i\in I}$ es una familia de conjuntos, la colección que
tiene por elementos a los elementos que pertenecen a
$A_i$ para toda $i$ en $I$ es un conjunto al que denotamos 
$\bigcap\{A_i\}_{i\in I}$ o $\underset{i\in I}{\bigcap} A_i$ y
llamamos \emph{la intersección de la familia $\{A_i\}_{i\in I}$}. 
\end{te}

Por ejemplo, si $I$ fuera el conjunto\footnote{Nota que este es
un conjunto si los alumnos son objetos primitivos} de los alumnos que
están inscritos en la clase de Álgebra Superior, para cada $i$ en
$I$, es decir, para cada alumno de la clase podemos considerar el
conjunto $A_i$ que tiene por elementos a los estados de la República
en los que ha estado. Entonces, la intersección de la familia
$\{A_i\}_{i\in I}$ es igual al conjunto de estados de la República en
los que han estado todos los alumnos del curso. Notemos que
este conjunto tiene al menos un elemento, el Distrito
Federal; y si algún alumno no ha estado en, digamos Yucatán, entonces
Yucatán no está en la intersección de la familia.  

\begin{te}
Si $\{A_i\}_{i\in I}$ es una familia de conjuntos, la colección que
tiene por elementos a los elementos que pertenecen a algún conjunto
$A_i$ para $i$ en $I$ es un conjunto al que denotamos
$\bigcup\{A_i\}_{i\in I}$ o $\underset{i\in I}{\bigcup} A_i$ y 
llamamos \emph{la unión de la familia $\{A_i\}_{i\in
I}$}. 
\end{te}

Por ejemplo, si $I$ fuera el conjunto de los alumnos que
están inscritos en la clase de Álgebra Superior, para cada $i$ en
$I$, es decir, para cada alumno de la clase podemos considerar el
conjunto $A_i$ que tiene por elementos a los lápices que trae el
alumno $i$ en la mochila. Entonces, la unión de la familia
$\{A_i\}_{i\in I}$ es igual al conjunto que tiene por elementos todos
los lápices que traen en la mochila cada uno de los alumnos en la
clase de Álgebra Superior.  

Las siguientes propiedades son conocidas como leyes de De Morgan
para familias de conjuntos. 

\begin{te} 
Si $E$ es un conjunto y $\{A_i\}_{i\in I}$ es una familia de conjuntos
entonces:
\begin{enumerate}
\item $E\dif \underset{i\in I}{\bigcup} A_i = \underset{i\in
I}{\bigcap} (E\dif A_i)$
\item $E\dif \underset{i\in I}{\bigcap} A_i = \underset{i\in 
I}{\bigcup} (E\dif A_i)$
\end{enumerate}
\end{te}

Nota que en particular estas igualdades son válidas para familias de
dos conjuntos, es decir, si $A$, $B$ y $E$ son tres conjuntos,
entonces:
\begin{enumerate}
\item  $E\dif (A\cup B) =(E\dif A)\cap (E\dif B)$
\item  $E\dif (A\cap B) =(E\dif A)\cup (E\dif B)$
\end{enumerate}

Las leyes de De Morgan nos permiten probar el \emph{principio de
dualidad} el cual dice que si tenemos una igualdad entre conjuntos que
involucre únicamente uniones e intersecciones, entonces también es
válida la igualdad dual, la cual se obtiene cambiando los símbolos de
unión por intersección y los símbolos de intersección por unión en la
igualdad original (ver ejercicio 12). Notese que  el principio de
dualidad es falso si en la igualdad original no intervienen solamente
uniones e intersecciones (ver ejercicios 13, 14 y 15).

\section{Ejercicios}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\renewcommand{\labelenumi}{\arabic{enumi}.}

\begin{enumerate}
\item Demuestra que si $A$ es un conjunto, la colección de todos los
objetos que no son elementos de $A$ no es un conjunto.
\item $A\cap B=A\dif (A\dif B)$. 
\item Si $A\cup B\subset C$, entonces $A\dif B=A\cap (C\dif B)$.
\item $A\subset B$ si y sólo si para todo conjunto $E$, $E\dif
B\subset E\dif A$.
\item $A\dif (B\dif C)=(A\dif B)\cup (A\cap C)$.
\item Muestra por medio de ejemplos que las siguientes proposiciones
son falsas:
 \begin{enumerate}
   \item $A\dif B=B\dif A$.
   \item Si $A\subset B\cup C$ entonces $A\subset B$ o $A\subset C$.
 \end{enumerate}
\newcommand{\tri}{\triangle}
\item Definimos la \emph{diferencia simétrica} de $A$ y $B$ como el
conjunto $(A\dif B)\cup (B\dif A)$ y lo denotamos por $A\tri
B$. Demuestra que:
 \begin{enumerate}
   \item $A\tri \emptyset=A$.
   \item $A\tri B=\emptyset$ si y sólo si $A=B$.
   \item Si $A\tri B= A\tri C$, entonces $B=C$.
 \end{enumerate}
\item $A\times B=\emptyset$ si y sólo si $A=\emptyset$ o
$B=\emptyset$.
\item $A\times B=B\times A$ si y sólo si $A=B$.
\item $A\times (B\cup C)=(A\times B)\cup (B\times C)$.
\item $A\times (B\cap C)=(A\times B)\cap (B\times C)$.
\item Usando las leyes de De Morgan y el ejercicio 4 enuncia y
demuestra las igualdades duales de los siguientes enunciados:
 \begin{enumerate}
   \item $A\cup(B\cup C)=(A\cup B)\cup C$.
   \item $A\cap\big(\underset{i\in I}{\bigcup}B_i\big)= 
          \underset{i\in I}{\bigcup}(A\cap B_i)$
 \end{enumerate}
\item $\po{A} \cap \po{B}=\po{A\cap B}$
\item $\po{A} \cup \po{B}\subset\po{A\cup B}$ 
\item Muestra con un ejemplo que la igualdad en el ejercicio 14 no
siempre se cumple. 
\end{enumerate}
